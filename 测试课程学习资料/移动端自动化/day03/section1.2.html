1.pytest的setup函数和teardown函数

1.1 概述

1.setup和teardown主要分为：模块级,类级，功能级，函数级。
2.存在于测试类内部

1.2 函数级别setUp() / teardown()

运行测试方法的始末,即运行一次测试函数运行一次setup和teardown

示例代码:
import pytest
class Test_ABC:

    def setup(self):
        print("------->setup_method")
    def teardown(self):
        print("------->teardown_method")
    def test_a(self):
        print("------->test_a")
    assert 1
    def test_b(self):
        print("------->test_b")

if __name__ == '__main__':
    pytest.main("-s test_02.py")

1.3 类级别

运行于测试类的始末,即:在一个测试内只运行一次setup_class和teardown_class，不关心测试类 内有多少个测试函数.

示例代码:
import pytest
class Test_ABC:
    # 测试类级开始
    def setup_class(self):
        print("------->setup_class")
    # 测试类级结束
    def teardown_class(self):
        print("------->teardown_class")
    def test_a(self):
        print("------->test_a")
    assert 1
    def test_b(self):
        print("------->test_b")
if __name__ == '__main__':
    pytest.main("-s test_03.py")

# 从appium库里面导入driver对象
from appium import webdriver
from time import sleep
class Test_002:
    def setup_class(self):
        import time,os
        # server 启动参数
        desired_caps = {}
        # 设备信息
        desired_caps['platformName'] = 'Android' #平台名称
        desired_caps['platformVersion'] = '5.1'  #平台版本
        desired_caps['deviceName'] = '192.168.56.101:5555' #设备号
        # app信息
        desired_caps['appPackage'] = 'com.android.settings' #应用的包名
        desired_caps['appActivity'] = '.Settings' #代表启动的activity
        self.driver = webdriver.Remote('http://localhost:4723/wd/hub', desired_caps) #声明driver对象

    def teardown_class(self):
        sleep(5)
        self.driver.quit()

    def test_set_location(self):
        """测试过程和断言"""
        # 存储
        save = self.driver.find_element_by_xpath("//*[contains(@text,'存储')]")
        # 更多
        more = self.driver.find_element_by_xpath("//*[contains(@text,'更多')]")
        # 存储 -> 更多
        self.driver.drag_and_drop(save, more)
        # 点击位置信息
        self.driver.find_element_by_xpath("//*[contains(@text,'位置')]").click()
        # 点击模式
        self.driver.find_element_by_xpath("//*[contains(@text,'模式')]").click()
        # 点击准确度高
        self.driver.find_element_by_xpath("//*[contains(@text,'使用GPS、')]").click()
        # 点击返回 按钮
        self.driver.find_element_by_class_name("android.widget.ImageButton").click()
        # # 获取结果 - 第一种断言方式
        res_text = self.driver.find_element_by_id("android:id/summary").text
        # # 断言
        assert "准确度高" == res_text
